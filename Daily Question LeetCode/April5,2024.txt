// Make The String Great
 
class Solution {
    public String makeGood(String s) {
           int i=1;
           while(i<s.length()){
               if(i>0 && Math.abs(s.charAt(i-1)-s.charAt(i))==32){
                   s = s.substring(0,i-1)+s.substring(i+1);
                   i--;
               }
               else{
                i++;
               }
           }
      return s;
        
    }
}
